import streamlit as st import pandas as pd import plotly.express as px import re # 1. Cấu hình trang Dashboard st.set_page_config(page_title="Dashboard Quản Lý Tài Xế", layout="wide", page_icon="🚗") st.title("🚗 Dashboard Phân Tích Hiệu Suất Tài Xế") st.markdown("Ứng dụng hỗ trợ import dữ liệu vận hành từ file Excel/CSV mẫu để theo dõi KPI trực quan.") # Hàm bổ trợ chuyển đổi định dạng thời gian "7h33m" thành dạng số (Giờ) để tính toán def parse_time_to_hours(time_str): if pd.isna(time_str) or not isinstance(time_str, str): return 0.0 match = re.match(r'(?:(\d+)h)?(?:(\d+)m)?', time_str) if match: hours = int(match.group(1)) if match.group(1) else 0 minutes = int(match.group(2)) if match.group(2) else 0 return hours + minutes / 60.0 return 0.0 # 2. Thành phần Upload File uploaded_file = st.file_uploader("Kéo thả hoặc chọn file dữ liệu tài xế (Excel hoặc CSV)", type=["xlsx", "csv"]) if uploaded_file is not None: try: # Đọc file tùy theo định dạng đầu vào if uploaded_file.name.endswith('.csv'): df = pd.read_csv(uploaded_file) else: df = pd.read_excel(uploaded_file) st.success("Tải dữ liệu thành công!") # 3. Tiền xử lý dữ liệu để vẽ biểu đồ và tính toán numeric_cols = ['Doanh số', 'Tổng giá trị cuốc xe', 'Số cuốc xe hoàn thành', 'Tỷ lệ nhận chuyến', 'Tỷ lệ hủy chuyến', 'Tỷ lệ hoàn thành'] for col in numeric_cols: if col in df.columns: df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0) if 'Tổng thời gian online' in df.columns: df['Thời gian online (Giờ)'] = df['Tổng thời gian online'].apply(parse_time_to_hours) # 4. Thanh điều hướng bên hông (Sidebar) - Bộ lọc st.sidebar.header("Bộ Lọc Dữ Liệu") # Lọc theo Khu vực khu_vuc_list = ['Tất cả'] + list(df['Khu vực'].dropna().unique()) if 'Khu vực' in df.columns else ['Tất cả'] selected_khu_vuc = st.sidebar.selectbox("Chọn Khu vực", khu_vuc_list) # Lọc theo Model xe model_list = ['Tất cả'] + list(df['Model'].dropna().unique()) if 'Model' in df.columns else ['Tất cả'] selected_model = st.sidebar.selectbox("Chọn Model Xe", model_list) # Lọc theo Tìm kiếm tên / mã tài xế search_driver = st.sidebar.text_input("Tìm kiếm Tên hoặc Mã tài xế") # Áp dụng bộ lọc vào Dataframe df_filtered = df.copy() if selected_khu_vuc != 'Tất cả': df_filtered = df_filtered[df_filtered['Khu vực'] == selected_khu_vuc] if selected_model != 'Tất cả': df_filtered = df_filtered[df_filtered['Model'] == selected_model] if search_driver: df_filtered = df_filtered[ df_filtered['Tên tài xế'].astype(str).str.contains(search_driver, case=False) | df_filtered['Mã tài xế'].astype(str).str.contains(search_driver, case=False) ] # 5. HIỂN THỊ CÁC CHỈ SỐ KPI TỔNG QUAN st.subheader("📊 Chỉ số KPI Tổng Quan") kpi1, kpi2, kpi3, kpi4 = st.columns(4) total_drivers = df_filtered['Mã tài xế'].nunique() total_revenue = df_filtered['Doanh số'].sum() total_trips = df_filtered['Số cuốc xe hoàn thành'].sum() avg_completion = df_filtered['Tỷ lệ hoàn thành'].mean() if 'Tỷ lệ hoàn thành' in df_filtered.columns else 0 kpi1.metric(label="Tổng số tài xế đang lọc", value=f"{total_drivers:,}") kpi2.metric(label="Tổng Doanh Số (VND)", value=f"{total_revenue:,.0f} đ") kpi3.metric(label="Tổng số cuốc hoàn thành", value=f"{total_trips:,} cuốc") kpi4.metric(label="Tỷ lệ hoàn thành trung bình", value=f"{avg_completion:.1f}%") st.markdown("---") # 6. HIỂN THỊ CÁC BIỂU ĐỒ PHÂN TÍCH (Vùng đồ thị chia làm 2 cột) st.subheader("📈 Biểu đồ Phân Tích Trực Quan") col1, col2 = st.columns(2) with col1: st.write("**Top 10 Tài Xế Có Doanh Số Cao Nhất**") if 'Tên tài xế' in df_filtered.columns and 'Doanh số' in df_filtered.columns: top_drivers = df_filtered.groupby('Tên tài xế')['Doanh số'].sum().reset_index() top_drivers = top_drivers.sort_values(by='Doanh số', ascending=False).head(10) fig_revenue = px.bar(top_drivers, x='Doanh số', y='Tên tài xế', orientation='h', labels={'Doanh số': 'Doanh số (VND)', 'Tên tài xế': 'Tài xế'}, color='Doanh số', color_continuous_scale='Blues') fig_revenue.update_layout(yaxis={'categoryorder':'total ascending'}) st.plotly_chart(fig_revenue, use_container_width=True) with col2: st.write("**Tỷ Lệ Đóng Góp Cuốc Xe Theo Model Xe**") if 'Model' in df_filtered.columns and 'Số cuốc xe hoàn thành' in df_filtered.columns: model_trips = df_filtered.groupby('Model')['Số cuốc xe hoàn thành'].sum().reset_index() fig_model = px.pie(model_trips, values='Số cuốc xe hoàn thành', names='Model', hole=0.4, color_discrete_sequence=px.colors.sequential.RdBu) st.plotly_chart(fig_model, use_container_width=True) col3, col4 = st.columns(2) with col3: st.write("**Phân Tích Tương Quan: Tỷ lệ Hoàn thành vs Tỷ lệ Hủy chuyến**") if all(c in df_filtered.columns for c in ['Tên tài xế', 'Tỷ lệ hoàn thành', 'Tỷ lệ hủy chuyến']): fig_scatter = px.scatter(df_filtered, x='Tỷ lệ hoàn thành', y='Tỷ lệ hủy chuyến', hover_data=['Tên tài xế', 'Mã tài xế'], color='Model', labels={'Tỷ lệ hoàn thành': 'Tỷ lệ hoàn thành (%)', 'Tỷ lệ hủy chuyến': 'Tỷ lệ hủy chuyến (%)'}) st.plotly_chart(fig_scatter, use_container_width=True) with col4: st.write("**Thời Gian Online Trung Bình Theo Khu Vực (Giờ)**") if 'Khu vực' in df_filtered.columns and 'Thời gian online (Giờ)' in df_filtered.columns: kv_online = df_filtered.groupby('Khu vực')['Thời gian online (Giờ)'].mean().reset_index() fig_kv = px.bar(kv_online, x='Khu vực', y='Thời gian online (Giờ)', labels={'Thời gian online (Giờ)': 'Thời gian Online TB (Giờ)'}, color='Khu vực', color_discrete_sequence=px.colors.qualitative.Pastel) st.plotly_chart(fig_kv, use_container_width=True) st.markdown("---") # 7. BẢNG DỮ LIỆU CHI TIẾT st.subheader("📋 Chi tiết Bảng Dữ Liệu (Sau khi lọc)") display_cols = [c for c in ['Mã tài xế', 'Tên tài xế', 'Biển kiểm soát', 'Model', 'Khu vực', 'Doanh số', 'Số cuốc xe hoàn thành', 'Tỷ lệ hoàn thành', 'Tổng thời gian online'] if c in df_filtered.columns] st.dataframe( df_filtered[display_cols].style.format({ 'Doanh số': '{:,.0f} đ', 'Tỷ lệ hoàn thành': '{:.1f}%' }), use_container_width=True ) except Exception as e: st.error(f"Đã xảy ra lỗi khi xử lý file dữ liệu: {e}") else: st.info("💡 Vui lòng kéo thả hoặc click chọn file dữ liệu tài xế (Excel hoặc CSV) ở trên để hiển thị Dashboard.")
